32 void InitWorkingSet(WorkingSet<PADDING>* toInit, bool sequential, size_t workingSetSize)
33 {
34  assert(toInit);
35 
36  // Initialising variables
37  size_t size = workingSetSize / sizeof(DummyStruct<PADDING>);
38 
39  DummyStruct<PADDING>* arr = (DummyStruct<PADDING>*) std::calloc(size, sizeof(*arr));
40  if (arr == nullptr)
41  {
42  throw std::invalid_argument("InitWorkingSet(): Unable to allocate working set");
43  }
44 
45  // Setting the 'next' links in the array
46  if (sequential)
47  {
48  for (size_t i = 0; i < size; ++i)
49  arr[i].next = &(arr[(i + 1) % size]);
50  }
51  else
52  {
53  size_t* indexes = (size_t*) std::calloc(size, sizeof(size_t));
54  for (size_t i = 0; i < size; indexes[i] = i, ++i);
55 
56  std::random_device rd;
57  std::mt19937 gen(rd());
58 
59  // It's takes some time, but i don't care
60  std::shuffle(&indexes[0], &(indexes[size - 1]), gen);
61 
62  for (size_t i = 0; i < size; ++i)
63  arr[indexes[i]].next = &(arr[indexes[(i + 1) % size]]);
64 
65  std::free(indexes);
66  }
67 
68  toInit->arr = arr;
69  toInit->size = size;
70 }
